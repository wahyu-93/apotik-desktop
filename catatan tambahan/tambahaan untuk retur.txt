// koreksi

CREATE TABLE tbl_log_koreksi (
  id            INT PRIMARY KEY AUTO_INCREMENT,
  no_koreksi    VARCHAR(50)  NOT NULL,           -- misal: KRK-1204202600001
  obat_id       INT          NOT NULL,
  batch_id      INT          NOT NULL,           -- batch yang dikoreksi
  stok_sebelum  INT          NOT NULL,           -- jumlah_sisa sebelum koreksi
  stok_sesudah  INT          NOT NULL,           -- jumlah_sisa setelah koreksi
  selisih       INT          NOT NULL,           -- sesudah - sebelum (bisa minus)
  alasan_kode   VARCHAR(50)  NOT NULL,           -- dropdown
  alasan_detail VARCHAR(200) NULL,                -- bebas ketik
  user_id       INT          NOT NULL,
  created_at    TIMESTAMP    DEFAULT CURRENT_TIMESTAMP,

  FOREIGN KEY (obat_id)  REFERENCES tbl_obat(id),
  FOREIGN KEY (batch_id) REFERENCES tbl_batch(id),
  FOREIGN KEY (user_id)  REFERENCES tbl_user(id)
);

//cek yg ada stoknya tapi belum ada di batch
// cekdulu 
SELECT 
  m.id,
  m.nama_obat,
  m.stok AS stok_master
FROM tbl_obat m
LEFT JOIN tbl_batch b ON b.obat_id = m.id
WHERE b.id IS NULL;

//insert yg masih belum ada
INSERT INTO tbl_batch (obat_id, no_batch, tgl_expired, jumlah_awal, jumlah_sisa, status, is_migrasi)
SELECT 
    m.id,
    CONCAT('STOK-AWAL-', m.id),
    m.tgl_exp,       
    m.stok,             
    m.stok,            
    1,                  
    1                  
FROM tbl_obat m
WHERE m.stok > 0
  AND NOT EXISTS (
      SELECT 1 FROM tbl_batch b WHERE b.obat_id = m.id
  );
